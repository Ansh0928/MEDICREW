**Extend streamConsultation yield shape:**

Currently `streamConsultation` yields `{ step, data }`. Extend to also yield agent identity at each node:

```typescript
export interface StreamEvent {
  step: string;
  data: Partial<ConsultationState>;
  agentName?: string;
  agentRole?: string;
  specialty?: string;
  eventType?: "node_output" | "token_delta" | "routing" | "complete";
  delta?: string; // Token-level text chunk for progressive streaming
}
```

**Implement token-level streaming (CONS-02):**

This is the critical change. Currently each agent node calls `llm.invoke()` which returns the full response at once. Switch to `llm.stream()` so the patient sees text appear progressively:

1. In each agent node function (triage, gp, specialist, recommend), replace `const response = await llm.invoke(messages)` with a streaming loop:

```typescript
// BEFORE (all-at-once):
const response = await llm.invoke(messages);
return { messages: [{ role, agentName, content: response.content }] };

// AFTER (token-level streaming via the generator):
const stream = await llm.stream(messages);
let fullContent = '';
for await (const chunk of stream) {
  const tokenText = typeof chunk.content === 'string' ? chunk.content : '';
  if (tokenText) {
    fullContent += tokenText;
    yield {
      step: nodeName,
      data: {},
      agentName: agent?.name,
      agentRole: agent?.role,
      specialty: agent?.specialties?.[0],
      eventType: 'token_delta' as const,
      delta: tokenText,
    };
  }
}
// After stream completes, yield the full node_output event with complete content
```

2. This requires restructuring the generator: instead of iterating over `graph.stream()` events and yielding once per node, each node's LLM call must itself yield token deltas. The recommended approach:
   - Keep `graph.stream()` for node orchestration (which node runs next)
   - Inside each node, use a callback or shared channel to emit token deltas
   - The simplest LangGraph-compatible approach: use `graph.streamEvents()` (v2) which emits `on_llm_stream` events with token chunks. Parse these events in `streamConsultation`:

```typescript
for await (const event of graph.streamEvents(initialState, { version: 'v2', ...streamConfig })) {
  if (event.event === 'on_llm_stream') {
    // Token-level delta from whichever agent node is currently running
    const chunk = event.data?.chunk;
    const tokenText = typeof chunk?.content === 'string' ? chunk.content : '';
    if (tokenText) {
      const nodeName = event.metadata?.langgraph_node;
      const agent = agentRegistry[nodeName as AgentRole];
      yield {
        step: nodeName || 'unknown',
        data: {},
        agentName: agent?.name,
        agentRole: agent?.role,
        specialty: agent?.specialties?.[0],
        eventType: 'token_delta',
        delta: tokenText,
      };
    }
  } else if (event.event === 'on_chain_end' && event.metadata?.langgraph_node) {
    // Full node output — emit as node_output with complete data
    const nodeName = event.metadata.langgraph_node;
    const agent = agentRegistry[nodeName as AgentRole];
    const nodeOutput = event.data?.output;

    yield {
      step: nodeName,
      data: nodeOutput as Partial<ConsultationState>,
      agentName: agent?.name,
      agentRole: agent?.role,
      specialty: agent?.specialties?.[0],
      eventType: 'node_output',
    };

    // After triage, emit routing event
    if (nodeName === 'triage') {
      const triageData = nodeOutput as Partial<ConsultationState>;
      if (triageData.relevantSpecialties && triageData.relevantSpecialties.length > 0) {
        yield {
          step: 'routing',
          data: { relevantSpecialties: triageData.relevantSpecialties },
          eventType: 'routing',
          agentName: 'MediCrew Coordinator',
          agentRole: 'orchestrator',
          specialty: 'Routing',
        };
      }
    }
  }
}
```

**AgentOverlay component** (`src/components/consult/AgentOverlay.tsx`):

Client component that displays the currently active agent during streaming:

```typescript
'use client';
import { CARE_TEAM } from '@/lib/care-team-config';
import { Badge } from '@/components/ui/badge';

interface AgentOverlayProps {
  agentName?: string;
  agentRole?: string;
  specialty?: string;
  isStreaming: boolean;
}

export function AgentOverlay({ agentName, agentRole, specialty, isStreaming }: AgentOverlayProps) {
  if (!agentName || !isStreaming) return null;

  const agent = CARE_TEAM.find(a => a.role === agentRole);
  const emoji = agent?.emoji || '';

  return (
    <div className="flex items-center gap-2 p-2 rounded-lg bg-muted/50 mb-2 animate-in fade-in">
      <span className="text-2xl">{emoji}</span>
      <div>
        <p className="font-medium text-sm">{agentName}</p>
        {specialty && <Badge variant="secondary" className="text-xs">{specialty}</Badge>}
      </div>
      <span className="ml-auto text-xs text-muted-foreground animate-pulse">Speaking...</span>
    </div>
  );
}
```
